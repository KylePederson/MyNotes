String Commands

Term Syntax Description
CONCAT SELECT CONCAT(column_1, "-", column_2, "-", column_3) FROM db_name; concatenates multiple specified values
CONCAT_WS SELECT CONCAT_WS("-", column_1, column_2, column_3) FROM db_name; concatenates multiple values together with a separator
SUBSTRING SELECT SUBSTRING('Hello World', 1, 5); Extracts a portion of the string by start location and number of characters
SUBSTRING SELECT SUBSTRING('Hello World', 7); Extracts a portion of the string by start location and going to the end of the string
SUBSTRING SELECT SUBSTRING('Hello World', -3); Extracts a portion of the string starting from the last character and going backward
SUBSTRING SELECT CONCAT(SUBSTRING(column_1, 1, 10), '...')FROM db_name; Extracts a portion of the string by start location and number of characters
REPLACE SELECT REPLACE('Hello World', 'l', '7'); replaces the specified value with a new value
CHAR_LENGTH SELECT CONCAT(column_1, ' is ', CHAR_LENGTH(column_1), ' characters long') FROM db_name; returns the number of characters in a string
UPPER SELECT CONCAT('THIS IS UPPER', UPPER(column_1)) FROM db_name; converts all characters to uppercase
LOWER SELECT CONCAT('this is lower ', LOWER(column_1)) FROM db_name; converts all characters to lowercase
DISTINCT SELECT DISTINCT CONCAT(column_1,' ', column_2) FROM db_name; Returns only the unique values for the specified column
ORDER BY SELECT column_1, column_2, column_3 FROM db_name ORDER BY column_2; Orders the results based on specified column
ORDER BY SELECT column_1, column_2, column_3 FROM db_name ORDER BY 2; # value specifies the column to order by
ORDER BY SELECT column_1, column_2, column_3 FROM db_name ORDER BY 1 DESC; DESC - sorts in descending order
ORDER BY SELECT column_1, column_2, column_3 FROM db_name ORDER BY 1 ASC; ASC - sorts in ascending order
LIMIT SELECT column_1 FROM db_name LIMIT 10; Specifies the max number of results to return
LIKE SELECT column_1, column_2 FROM db_name WHERE column_1 LIKE '%da%'; % - Wildcard
LIKE SELECT column_1, column_2 FROM db_name WHERE column_2 LIKE '____'; _ - Wildcard to find any ONE character
LIKE SELECT column_1 FROM db_name WHERE column_1 LIKE '%\_%'; \ - Nullifies the next key phrase
