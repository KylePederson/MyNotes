Table Commands

Term Syntax Description
CREATE TABLE CREATE TABLE table_name(column_1 Data_Type (Parameter), column_2 Data_Type (Parameter)); Creates a Table in the currently selected database
DROP TABLE DROP TABLE table_name; Permanently removes a table form the database
NOT NULL CREATE TABLE table_name (column_1 Data_Type (Parameter) NOT NULL, column_2 Data_Type (Parameter) NOT NULL); NOT NULL - Specifies that a table value cannot be NULL
DEFAULT CREATE TABLE table_name (column_1 Data_Type (Parameter) DEFAULT 'unnamed', column_2 Data_Type (Parameter) DEFAULT 99); DEFAULT - if no table value is given, then the specified default is used
PRIMARY KEY() CREATE TABLE table_name (column_1 Data_Type (Parameter), column_2 Data_Type (Parameter), PRIMARY KEY (column_1)); PRIMARY KEY - Specifies that this column value must be unique
AUTO_INCREMENT CREATE TABLE table_name (column_1 INT AUTO_INCREMENT, column_2 Data_Type (Parameter)); AUTO_INCREMENT - Automatically increments the integer after each new entry
SHOW COLUMNS SHOW COLUMNS FROM table_name; Description
DESC DESC table_name; Description
SELECT * FROM SELECT * FROM table_name; Select all columns from a table
SELECT [col] FROM SELECT column_1, column_2, column_3 FROM table_name; allows the selection of multiple columns from a table
WHERE SELECT * FROM table_name WHERE column_1 = 'value'; WHERE - allows the selection of specific data based on given criteria
AS SELECT column_1 AS new_name FROM table_name; AS - allows the data to be displayed differently without changing the database
SHOW TABLES SHOW TABLES; Description
SHOW WARNINGS SHOW WARNINGS; SHOW WARNINGS displays all warnings if one is encountered
INSERT INTO INSERT INTO table_name (column_1, column_2) Values ('Value 1', 'Value 2'), ('Value 1', 'Value 2'); INSERT INTO - Add values into a table by column name and data type
UPDATE UPDATE table_name SET column_1 = 'new_value' WHERE column_1 = 'current_value'; UPDATE - modify an existing value within a table
DELETE DELETE FROM table_name WHERE column_1 = 'current_value'; Description
