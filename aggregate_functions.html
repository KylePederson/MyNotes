Aggregate Functions
Term Syntax Description
COUNT SELECT COUNT(*)
FROM tablename; 		Count of specified value
GROUP BY SELECT column_1, column_2
FROM tablename
GROUP BY column_2; 		Groups similar values together
MIN SELECT MIN (column_1)
FROM tablename; 		Returns the smallest value
MAX SELECT MAX (column_1)
FROM tablename;		 Returns the largest value
SUM SELECT column_1, column_2,
SUM (column_3)
FROM tablename
GROUP BY column_1, column_2; 		--DESCRIPTION PLACEHOLDER--
AVG SELECT column_1, column_2,
AVG (column_3)
FROM tablename
GROUP BY column_1, column_2; 		--DESCRIPTION PLACEHOLDER--
SUB QUERIES SELECT column_1, column_2
FROM tablename
